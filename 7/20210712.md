// common_func.js 파일에 함수 작성.

// 이메일 유효성 체크 

function isValidEmail( email ){
  // 이메일 문자열을 체크하는 RegExp 객체 생성하기.
  var regExp = new RegExp(/^([0-9a-zA-Z_-]+)@([0-9a-zA-Z_-]+)(\.[0-9a-zA-Z_-]+){1,2}$/);
  return regExp.test(email);
}
