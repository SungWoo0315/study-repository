  ```javascript

  // 누적시켜서 출력하는 다른 방법. Array 사용해서 배열안에 누적시켜서 출력하기.  
  // -------------------------------------------------
  // common_date.js 파일 안 scheduleMsg 변수안의 2차원 Array 객체에 접근해서
  // 오늘 날짜와 일치하는 놈의 메시지들을 모아서 리턴하는 함수 
  // -------------------------------------------------
  function getScheduleMsg(){
      // ---------------------------
      // 메시지를 저장할 Array 객체 생성하기.
      // ---------------------------
      var arr = [];
      // ---------------------------
      // scheduleMsg 변수안의 2차원 Array 객체에 접근해서
      // 오늘날짜와 일치하는 놈의 메시지들을 모아서 Array 객체에 저장하기
      // ---------------------------
      for( var i=0; i<scheduleMsg.length; i++){
          var date = scheduleMsg[i][0];
          date = deleteBlank(date);  // 공백제거 공용함수 사용.  
          var msg = scheduleMsg[i][1];
          if( isToday(date) ){
              // alert( msg ) // 테스트 할때 보려고 alert 한것. 주석처리.
              arr.push(msg); // arr 안에 누적시키기. 
                             //특정날짜에 메시지가 2개 이상 나올 경우 누석시켜 다 나오게 하기.
          }
      }	
      // ---------------------------
      // Array 객체에 저장된 메시지를 모두 붙여 문자로 리턴하기.
      // ---------------------------
      return arr.join(" <br><br> ");	// arr 에 join 사용해서 배열 붙여서 출력하기
                                      // 중간에 <br>을 넣어주어 이어 붙인 사이에 줄바꿈을 해준다.
  }

  ```

  ```javascript

  // -------------------------------------------------
  // 명언을 랜덤하게 나오게 하기.
  // 명언을 리턴하는 함수 선언.
  // 명언 문자열은 common_date.js 파일안 goldenSaying 변수안에 저장되어 있다.
  // -------------------------------------------------
  function get_goldenSaying(){
      // ---------------------------
      // goldenSaying 변수안에 저장된 Array 객체 안의 배열변수 개수 얻기
      // 이 개수가 명언의 개수이다.   
      // ---------------------------
      var goldenSaying_cnt = goldenSaying.length;
      // ---------------------------
      // 0 ~ 명언개수-1 사이의 무작위 인덱스 번호 얻기.
      // 이 인덱스번호 무작위 배열 변수를 선택할 때 사용된다.
      // ---------------------------
      var random_index = Math.floor( Math.random()*goldenSaying_cnt );
      // ---------------------------
      // 무작위 인덱스 번호에 해당하는 배열변수 안의 데이터를 리턴하기.
      // ---------------------------
      return goldenSaying[random_index];
  }

  ```

  ```javascript

  // -------------------------------------------------
  // D-Day 나오게 하기.
  // 오늘부터 특정 날까지 남은날이 10일 이하일 때 그 특정 날짜에 해당하는 메시지들을 리턴하는 함수.
  // 특정 날짜와 메시지는 common_date.js 파일안 dDayMsg 변수안에 저장되어 있다.
  // -------------------------------------------------
  function get_DDayMsg(){
      // ---------------------------
      // D-Day에 대응하는 메시지를 저장할 Array 객체 생성하기.
      // ---------------------------
      var arr = [];
      // ---------------------------
      // dDayMsg 변수안의 2차원 Array 객체에 접근해서
      // D-Day가 10일 이하일때의 메시지들을 모아서 Array 객체에 저장하기.
      // ---------------------------
      for( var i=0; i<dDayMsg.length; i++){
          var date = dDayMsg[i][0];
          date = deleteBlank(date);  // 공백제거 공용함수 사용.  
          var msg = dDayMsg[i][1];

          // ---------------------------
          // get_dDay 함수 호출하여 오늘부터 Array 객체 안의 i번째 날짜까지의 남은 일수 구해서
          // 변수 dDay 에 저장하기.
          // ---------------------------
          var dDay = get_dDay(date);

          // ---------------------------
          // 구한 dDay 가 0~10 일 사이면 Array 객체안에 저장.
          // ---------------------------
          if( dDay>=0 && dDay<=10 ){
              arr.push(date + " 까지 " + msg + " 은(는) " + dDay + " 일 남았습니다."); // arr 안에 누적시키기. 
                              //특정날짜에 메시지가 2개 이상 나올 경우 누석시켜 다 나오게 하기.
          }
      }
      // ---------------------------
      // Array 객체에 저장된 메시지들이 1개 이상이면 모든 메시지를 붙여 리턴하기.
      // ---------------------------
      if (arr.length>0){
          return arr.join(" <br> ");	
      }
      // ---------------------------
      // Array 객체에 저장된 메시지들이 0개면 "" 리턴하기.
      // ---------------------------
      else{
          return "";
      }	
  }

  ```
